@app.route('/verify_mikrotik_password', methods=['POST']) def verify_mikrotik_password(): """Vérifie le mot de passe d'un Mikrotik""" try: data = request.json mik_id = data.get('mik_id') account_key = data.get('account_key') password = data.get('password') if not all([mik_id, account_key, password]): return jsonify({"success": False, "message": "Champs requis"}), 400 conn = get_db() if not conn: return jsonify({"success": False, "message": "Erreur base de données"}), 500 cursor = conn.cursor(dictionary=True) cursor.execute(""" SELECT mik_password FROM mikrotik WHERE mik_id = %s AND account_key = %s """, (mik_id, account_key)) mik = cursor.fetchone() cursor.close() conn.close() if not mik: return jsonify({"success": False, "message": "Mikrotik non trouvé"}), 404 if mik['mik_password'] == password: return jsonify({"success": True, "message": "Mot de passe correct"}), 200 else: return jsonify({"success": False, "message": "Mot de passe incorrect"}), 401 except Exception as e: logger.error(f"Erreur vérification mot de passe: {e}") return jsonify({"success": False, "message": str(e)}), 500
